713
714template<class T>
715void Visitor::dispatch(const T& obj) {
716 auto hash = reinterpret_cast<size_t>(&obj);
717 if (visited_.find(hash) != std::end(visited_)) {
718 // Already visited
719 return;
720 }
721
722 visited_.insert(hash);
723 visit(obj);
724}
