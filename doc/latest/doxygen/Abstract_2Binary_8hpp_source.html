180 template<class T>
181 LIEF::result<T> get_int_from_virtual_address(
182 uint64_t va, VA_TYPES addr_type = VA_TYPES::AUTO) const
183 {
184 T value;
185 static_assert(std::is_integral<T>::value, "Require an integral type");
186 span<const uint8_t> raw = get_content_from_virtual_address(va, sizeof(T), addr_type);
187 if (raw.empty() || raw.size() < sizeof(T)) {
188 return make_error_code(lief_errors::read_error);
189 }
190
191 std::copy(raw.data(), raw.data() + sizeof(T),
192 reinterpret_cast<uint8_t*>(&value));
193 return value;
194 }
